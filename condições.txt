-| WHILE:
------------------------------------------------

  command1
  command2
done

-| UNTIL:
------------------------------------------------
VAR=1
COUNT=0
until [ $VAR = "0" ]; do
  command1
  command2
  if [ $COUNT = 9 ]; then
    VAR=0
  fi
  COUNT=`expr $COUNT + 1`
done
